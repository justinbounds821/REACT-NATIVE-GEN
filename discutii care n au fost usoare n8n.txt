const fs = require('fs');

  const dir = path.dirname(filePath);

    fs.mkdirSync(dir, { recursive: true });

const raw = $json["body"];
const lines = raw.split('\n');
const files = [];

let currentFile = null;

for (let line of lines) {
  const match = line.match(/^📄 (.+?):$/);
  if (match) {
    if (currentFile && buffer.length > 0) {
      files.push({
        filename: currentFile,
        content: buffer.join('\n').trim()
      });
    }
    currentFile = match[1].trim();
    buffer = [];
  } else {
    if (currentFile) {
      buffer.push(line);
    }
  }
}

// Push final file
if (currentFile && buffer.length > 0) {
  files.push({
    filename: currentFile,
    content: buffer.join('\n').trim()
  });
}

return files.map(f => ({
  json: {
    filename: f.filename,
    content: f.content
  }
}));

const baseDir = '/mnt/data/generated_app';  // poți schimba unde să salveze

  const filename = item.json.filename;

  // Redirecționăm în subfoldere după reguli

  let relativePath = filename;
  let matchedFolder = null;

  // 🧭 Detectează folderul de destinație
  for (const prefix in folderMap) {
    if (filename.startsWith(prefix)) {

from docx import Document
from docx.shared import Pt
import re
import os

# Creează documentul
doc = Document()
doc.add_heading('📦 Cod generat de Claude', 0)

# Preia textul generat de Claude
raw = items[0]['json'].get('claude_output', '')

# Pattern pentru extragere fișiere `📄 filename:` + blocuri de cod în ``` ```
pattern = r"📄 ([^\n]+):\s*```[a-z]*\n(.*?)```"
matches = re.findall(pattern, raw, re.DOTALL)

# Formatare frumoasă pentru fiecare fișier
for filename, code in matches:
    doc.add_heading(filename.strip(), level=2)
    para = doc.add_paragraph()
    run = para.add_run(code.strip())
    run.font.name = 'Courier New'
    run.font.size = Pt(9)

# Creează folderul dacă nu există
output_path = "/mnt/data/exported_code.docx"
os.makedirs(os.path.dirname(output_path), exist_ok=True)

# Salvează documentul
doc.save(output_path)

# Returnează path-ul
return [{
    "json": {
        "status": "✅ Export DOCX complet",
        "output_path": output_path,
        "files_exported": len(matches)
    }
}]
